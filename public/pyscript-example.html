import re

main_text_element = Element("main_text")
emails_element = Element("emails")

def extract_emails(*args, **kwargs):
  text = main_text_element.value
  emails = ', '.join(re.findall(r"[a-z0-9\.\-+_]+@[a-z0-9\.\-+_]+\.[a-z]+", text))
  emails_element.write(emails)
